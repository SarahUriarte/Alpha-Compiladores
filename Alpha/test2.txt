let
    var x : int;
    var n : int;
    var res : int;
    var cadena: string
in
    begin
        x := 5;
        n := x;
        res := 1;
        while x>3 do
        begin
            res := res * n;
            n := n - 1;
            print(n)
        end;
        cadena := "factorial";
        print("Resultado del " + cadena + ": ");
        print(res)
    end
